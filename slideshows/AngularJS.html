ng-app

Declares the root element of an AngularJS application, under which directives can be used to declare bindings and define behavior.

The code

							
<!DOCTYPE html>
<html ng-app="app">
    ...
    var app = angular.module('app', []);

Then, you built an Angular application.

ng-controller

With Scope

							
<head>
	<script>
		app.controller('nameCtrl', function($scope) {
			$scope.name = {
				first: "Anson",
				last:  "Gao"
			};
		});
	</script>
</head>
<body ng-controller="nameCtrl">
	<h2>
		<span ng-bind="name.first"></span>
		<span ng-bind="name.last"></span>
	</h2>
</body>

You created a controller with $scope.

Preview

Without Scope

							
<head>
	<script>
		app.controller('nameCtrlClass', function() {
			this.first: "Anson";
			this.last:  "Gao";
		});
	</script>
</head>
<body ng-controller="nameCtrlClass as nameObj">
	<span ng-bind="nameObj.first"></span>
	<span ng-bind="nameObj.last"></span>
</body>

You created a controller as a class.

ng-if

How to use?

							
<head>
	<script>
		app.controller('nameCtrlClass', function() {
			this.first: "Anson";
			this.last:  "Gao";
		});
	</script>
</head>
<body ng-controller="nameCtrlClass as nameObj">
	<span ng-bind="nameObj.first" ng-if="nameObj.first == 'Anson'"></span>
	<span ng-bind="nameObj.last" ng-if="nameObj.last == 'Chen'"></span>
</body>

ng-repeat

How to use?

							
<head>
	<script>
		app.controller('renderList', function($scope) {
			$scope.memberList = [
				{firstName: "Rog", lastName: "Lee"},
				{firstName: "Derek", lastName: "Wu"},
				{firstName: "Anson", lastName: "Gao"},
				{firstName: "Larry", lastName: "Lo"}
			];
		});
	</script>
</head>
<body ng-controller="renderList">
	<p ng-repeat="member in memberList">
		<span ng-bind="member.firstName"></span>
		<span ng-bind="member.lastName"></span>
	</p>
</body>
